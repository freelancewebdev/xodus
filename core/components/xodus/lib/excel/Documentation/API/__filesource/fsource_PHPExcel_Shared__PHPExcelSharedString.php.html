  501.     /**
  502.      * Decode UTF-16 encoded strings.
  503.      *
  504.      * Can handle both BOM'ed data and un-BOM'ed data.
  505.      * Assumes Big-Endian byte order if no BOM is available.
  506.      * This function was taken from http://php.net/manual/en/function.utf8-decode.php
  507.      * and $bom_be parameter added.
  508.      *
  509.      * @param   string  $str  UTF-16 encoded data to decode.
  510.      * @return  string  UTF-8 / ISO encoded data.
  511.      * @access  public
  512.      * @version 0.2 / 2010-05-13
  513.      * @author  Rasmus Andersson {@link http://rasmusandersson.se/}
  514.      * @author vadik56
  515.      */
  516.     public static function utf16_decode( $str, $bom_be=true ) {
  517.         if( strlen($str) < 2 ) return $str;
  518.         $c0 = ord($str{0});
  519.         $c1 = ord($str{1});
  520.         if( $c0 == 0xfe && $c1 == 0xff ) { $str = substr($str,2); }
  521.         elseif( $c0 == 0xff && $c1 == 0xfe ) { $str = substr($str,2); $bom_be = false; }
  522.         $len = strlen($str);
  523.         $newstr = '';
  524.         for($i=0;$i<$len;$i+=2) {
  525.             if( $bom_be ) { $val = ord($str{$i})   << 4; $val += ord($str{$i+1}); }
  526.             else {        $val = ord($str{$i+1}) << 4; $val += ord($str{$i}); }
  527.             $newstr .= ($val == 0x228) ? "\n" : chr($val);
  528.         }
  529.         return $newstr;
  530.     }
