  546.                 return false;
  547.             }
  548.             // get the input sequence array
  549.             $sequence = $this->getInputSequences($code);
  550.             $codewords = array(); // array of code-words
  551.             foreach($sequence as $seq) {
  552.                 $cw = $this->getCompaction($seq[0], $seq[1], true);
  553.                 $codewords = array_merge($codewords, $cw);
  554.             }
  555.             if ($codewords[0] == 900) {
  556.                 // Text Alpha is the default mode, so remove the first code
  557.                 array_shift($codewords);
  558.             }
  559.             // count number of codewords
  560.             $numcw = count($codewords);
  561.             if ($numcw > 925) {
  562.                 // reached maximum data codeword capacity
  563.                 return false;
  564.             }
  565.             // build macro control block codewords
  566.             if (!empty($macro)) {
  567.                 $macrocw = array();
  568.                 // beginning of macro control block
  569.                 $macrocw[] = 928;
  570.                 // segment index
  571.                 $cw = $this->getCompaction(902, sprintf('%05d', $macro['segment_index']), false);
  572.                 $macrocw = array_merge($macrocw, $cw);
  573.                 // file ID
  574.                 $cw = $this->getCompaction(900, $macro['file_id'], false);
  575.                 $macrocw = array_merge($macrocw, $cw);
  576.                 // optional fields
  577.                 $optmodes = array(900,902,902,900,900,902,902);
  578.                 $optsize = array(-1,2,4,-1,-1,-1,2);
  579.                 foreach ($optmodes as $k => $omode) {
  580.                     if (isset($macro['option_'.$k])) {
  581.                         $macrocw[] = 923;
  582.                         $macrocw[] = $k;
  583.                         if ($optsize[$k] == 2) {
  584.                             $macro['option_'.$k] = sprintf('%05d', $macro['option_'.$k]);
  585.                         } elseif ($optsize[$k] == 4) {
  586.                             $macro['option_'.$k] = sprintf('%010d', $macro['option_'.$k]);
  587.                         }
  588.                         $cw = $this->getCompaction($omode, $macro['option_'.$k], false);
  589.                         $macrocw = array_merge($macrocw, $cw);
  590.                     }
  591.                 }
  592.                 if ($macro['segment_index'] == ($macro['segment_total'] - 1)) {
  593.                     // end of control block
  594.                     $macrocw[] = 922;
  595.                 }
  596.                 // update total codewords
  597.                 $numcw += count($macrocw);
  598.             }
  599.             // set error correction level
  600.             $ecl = $this->getErrorCorrectionLevel($ecl, $numcw);
  601.             // number of codewords for error correction
  602.             $errsize = (2 << $ecl);
  603.             // calculate number of columns (number of codewords per row) and rows
  604.             $nce = ($numcw + $errsize + 1);
  605.             $cols = round((sqrt(4761 + (68 * $aspectratio * ROWHEIGHT * $nce)) - 69) / 34);
  606.             // adjust cols
  607.             if ($cols < 1) {
  608.                 $cols = 1;
  609.             } elseif ($cols > 30) {
  610.                 $cols = 30;
  611.             }
  612.             $rows = ceil($nce / $cols);
  613.             $size = ($cols * $rows);
  614.             // adjust rows
  615.             if (($rows < 3) OR ($rows > 90)) {
  616.                 if ($rows < 3) {
  617.                     $rows = 3;
  618.                 } elseif ($rows > 90) {
  619.                     $rows = 90;
  620.                 }
  621.                 $cols = ceil($size / $rows);
  622.                 $size = ($cols * $rows);
  623.             }
  624.             if ($size > 928) {
  625.                 // set dimensions to get maximum capacity
  626.                 if (abs($aspectratio - (17 * 29 / 32)) < abs($aspectratio - (17 * 16 / 58))) {
  627.                     $cols = 29;
  628.                     $rows = 32;
  629.                 } else {
  630.                     $cols = 16;
  631.                     $rows = 58;
  632.                 }
  633.                 $size = 928;
  634.             }
  635.             // calculate padding
  636.             $pad = ($size - $nce);
  637.             if ($pad > 0) {
  638.                 if (($size - $rows) == $nce) {
  639.                     --$rows;
  640.                     $size -= $rows;
  641.                 } else {
  642.                     // add pading
  643.                     $codewords = array_merge($codewords, array_fill(0, $pad, 900));
  644.                 }
  645.             }
  646.             if (!empty($macro)) {
  647.                 // add macro section
  648.                 $codewords = array_merge($codewords, $macrocw);
  649.             }
  650.             // Symbol Lenght Descriptor (number of data codewords including Symbol Lenght Descriptor and pad codewords)
  651.             $sld = $size - $errsize;
  652.             // add symbol length description
  653.             array_unshift($codewords, $sld);
  654.             // calculate error correction
  655.             $ecw = $this->getErrorCorrection($codewords, $ecl);
  656.             // add error correction codewords
  657.             $codewords = array_merge($codewords, $ecw);
  658.             // add horizontal quiet zones to start and stop patterns
  659.             $pstart = str_repeat('0', QUIETH).$this->start_pattern;
  660.             $pstop = $this->stop_pattern.str_repeat('0', QUIETH);
  661.             $barcode_array['num_rows'] = ($rows * ROWHEIGHT) + (2 * QUIETV);
  662.             $barcode_array['num_cols'] = (($cols + 2) * 17) + 35 + (2 * QUIETH);
  663.             $barcode_array['bcode'] = array();
  664.             // build rows for vertical quiet zone
  665.             if (QUIETV > 0) {
  666.                 $empty_row = array_fill(0, $barcode_array['num_cols'], 0);
  667.                 for ($i = 0; $i < QUIETV; ++$i) {
  668.                     // add vertical quiet rows
  669.                     $barcode_array['bcode'][] = $empty_row;
  670.                 }
  671.             }
  672.             $k = 0; // codeword index
  673.             $cid = 0; // initial cluster
  674.             // for each row
  675.             for ($r = 0; $r < $rows; ++$r) {
  676.                 // row start code
  677.                 $row = $pstart;
  678.                 switch ($cid) {
  679.                     case 0: {
  680.                         $L = ((30 * intval($r / 3)) + intval(($rows - 1) / 3));
  681.                         break;
  682.                     }
  683.                     case 1: {
  684.                         $L = ((30 * intval($r / 3)) + ($ecl * 3) + (($rows - 1) % 3));
  685.                         break;
  686.                     }
  687.                     case 2: {
  688.                         $L = ((30 * intval($r / 3)) + ($cols - 1));
  689.                         break;
  690.                     }
  691.                 }
  692.                 // left row indicator
  693.                 $row .= sprintf('%17b', $this->clusters[$cid][$L]);
  694.                 // for each column
  695.                 for ($c = 0; $c < $cols; ++$c) {
  696.                     $row .= sprintf('%17b', $this->clusters[$cid][$codewords[$k]]);
  697.                     ++$k;
  698.                 }
  699.                 switch ($cid) {
  700.                     case 0: {
  701.                         $L = ((30 * intval($r / 3)) + ($cols - 1));
  702.                         break;
  703.                     }
  704.                     case 1: {
  705.                         $L = ((30 * intval($r / 3)) + intval(($rows - 1) / 3));
  706.                         break;
  707.                     }
  708.                     case 2: {
  709.                         $L = ((30 * intval($r / 3)) + ($ecl * 3) + (($rows - 1) % 3));
  710.                         break;
  711.                     }
  712.                 }
  713.                 // right row indicator
  714.                 $row .= sprintf('%17b', $this->clusters[$cid][$L]);
  715.                 // row stop code
  716.                 $row .= $pstop;
  717.                 // convert the string to array
  718.                 $arow = preg_split('//', $row, -1, PREG_SPLIT_NO_EMPTY);
  719.                 // duplicate row to get the desired height
  720.                 for ($h = 0; $h < ROWHEIGHT; ++$h) {
  721.                     $barcode_array['bcode'][] = $arow;
  722.                 }
  723.                 ++$cid;
  724.                 if ($cid > 2) {
  725.                     $cid = 0;
  726.                 }
  727.             }
  728.             if (QUIETV > 0) {
  729.                 for ($i = 0; $i < QUIETV; ++$i) {
  730.                     // add vertical quiet rows
  731.                     $barcode_array['bcode'][] = $empty_row;
  732.                 }
  733.             }
  734.             $this->barcode_array = $barcode_array;
  735.         }
  736.  
  737.         /**
  738.          * Returns a barcode array which is readable by TCPDF
  739.          * @return array barcode array readable by TCPDF;
  740.          * @access public
  741.          */
  742.         public function getBarcodeArray() {
  743.             return $this->barcode_array;
  744.         }
  745.  
  746.         /**
  747.          * Returns the error correction level (0-8) to be used
  748.          * @param int $ecl error correction level
  749.          * @param int $numcw number of data codewords
  750.          * @return int error correction level
  751.          * @access protected
  752.          */
  753.         protected function getErrorCorrectionLevel($ecl, $numcw) {
  754.             // get maximum correction level
  755.             $maxecl = 8; // starting error level
  756.             $maxerrsize = (928 - $numcw); // available codewords for error
  757.             while ($maxecl > 0) {
  758.                 $errsize = (2 << $ecl);
  759.                 if ($maxerrsize >= $errsize) {
  760.                     break;
  761.                 }
  762.                 --$maxecl;
  763.             }
  764.             // check for automatic levels
  765.             if (($ecl < 0) OR ($ecl > 8)) {
  766.                 if ($numcw < 41) {
  767.                     $ecl = 2;
  768.                 } elseif ($numcw < 161) {
  769.                     $ecl = 3;
  770.                 } elseif ($numcw < 321) {
  771.                     $ecl = 4;
  772.                 } elseif ($numcw < 864) {
  773.                     $ecl = 5;
  774.                 } else {
  775.                     $ecl = $maxecl;
  776.                 }
  777.             }
  778.             if ($ecl > $maxecl) {
  779.                 $ecl = $maxecl;
  780.             }
  781.             return $ecl;
  782.         }
  783.  
  784.         /**
  785.          * Returns the error correction codewords
  786.          * @param array $cw array of codewords including Symbol Lenght Descriptor and pad
  787.          * @param int $ecl error correction level 0-8
  788.          * @return array of error correction codewords
  789.          * @access protected
  790.          */
  791.         protected function getErrorCorrection($cw, $ecl) {
  792.             // get error correction coefficients
  793.             $ecc = $this->rsfactors[$ecl];
  794.             // number of error correction factors
  795.             $eclsize = (2 << $ecl);
  796.             // maximum index for $rsfactors[$ecl]
  797.             $eclmaxid = ($eclsize - 1);
  798.             // initialize array of error correction codewords
  799.             $ecw = array_fill(0, $eclsize, 0);
  800.             // for each data codeword
  801.             foreach($cw as $k => $d) {
  802.                 $t1 = ($d + $ecw[$eclmaxid]) % 929;
  803.                 for ($j = $eclmaxid; $j > 0; --$j) {
  804.                     $t2 = ($t1 * $ecc[$j]) % 929;
  805.                     $t3 = 929 - $t2;
  806.                     $ecw[$j] = ($ecw[($j - 1)] + $t3) % 929;
  807.                 }
  808.                 $t2 = ($t1 * $ecc[0]) % 929;
  809.                 $t3 = 929 - $t2;
  810.                 $ecw[0] = $t3 % 929;
  811.             }
  812.             foreach($ecw as $j => $e) {
  813.                 if ($e != 0) {
  814.                     $ecw[$j] = 929 - $e;
  815.                 }
  816.             }
  817.             $ecw = array_reverse($ecw);
  818.             return $ecw;
  819.         }
  820.  
  821.         /**
  822.          * Create array of sequences from input
  823.          * @param string $code code
  824.          * @return bidimensional array containing characters and classification
  825.          * @access protected
  826.          */
  827.         protected function getInputSequences($code) {
  828.             $sequence_array = array(); // array to be returned
  829.             $numseq = array();
  830.             // get numeric sequences
  831.             preg_match_all('/([0-9]{13,})/', $code, $numseq, PREG_OFFSET_CAPTURE);
  832.             $numseq[1][] = array('', strlen($code));
  833.             $offset = 0;
  834.             foreach($numseq[1] as $seq) {
  835.                 $seqlen = strlen($seq[0]);
  836.                 if ($seq[1] > 0) {
  837.                     // extract text sequence before the number sequence
  838.                     $prevseq = substr($code, $offset, ($seq[1] - $offset));
  839.                     $textseq = array();
  840.                     // get text sequences
  841.                     preg_match_all('/([\x09\x0a\x0d\x20-\x7e]{5,})/', $prevseq, $textseq, PREG_OFFSET_CAPTURE);
  842.                     $textseq[1][] = array('', strlen($prevseq));
  843.                     $txtoffset = 0;
  844.                     foreach($textseq[1] as $txtseq) {
  845.                         $txtseqlen = strlen($txtseq[0]);
  846.                         if ($txtseq[1] > 0) {
  847.                             // extract byte sequence before the text sequence
  848.                             $prevtxtseq = substr($prevseq, $txtoffset, ($txtseq[1] - $txtoffset));
  849.                             if (strlen($prevtxtseq) > 0) {
  850.                                 // add BYTE sequence
  851.                                 if ((strlen($prevtxtseq) == 1) AND ((count($sequence_array) > 0) AND ($sequence_array[(count($sequence_array) - 1)][0] == 900))) {
  852.                                     $sequence_array[] = array(913, $prevtxtseq);
  853.                                 } elseif ((strlen($prevtxtseq) % 6) == 0) {
  854.                                     $sequence_array[] = array(924, $prevtxtseq);
  855.                                 } else {
  856.                                     $sequence_array[] = array(901, $prevtxtseq);
  857.                                 }
  858.                             }
  859.                         }
  860.                         if ($txtseqlen > 0) {
  861.                             // add numeric sequence
  862.                             $sequence_array[] = array(900, $txtseq[0]);
  863.                         }
  864.                         $txtoffset = $txtseq[1] + $txtseqlen;
  865.                     }
  866.                 }
  867.                 if ($seqlen > 0) {
  868.                     // add numeric sequence
  869.                     $sequence_array[] = array(902, $seq[0]);
  870.                 }
  871.                 $offset = $seq[1] + $seqlen;
  872.             }
  873.             return $sequence_array;
  874.         }
  875.  
  876.         /**
  877.          * Compact data by mode.
  878.          * @param int $mode compaction mode number
  879.          * @param string $code data to compact
  880.          * @param boolean $addmode if true add the mode codeword at first position
  881.          * @return array of codewords
  882.          * @access protected
  883.          */
  884.         protected function getCompaction($mode, $code, $addmode=true) {
  885.             $cw = array(); // array of codewords to return
  886.             switch($mode) {
  887.                 case 900: { // Text Compaction mode latch
  888.                     $submode = 0; // default Alpha sub-mode
  889.                     $txtarr = array(); // array of characters and sub-mode switching characters
  890.                     $codelen = strlen($code);
  891.                     for ($i = 0; $i < $codelen; ++$i) {
  892.                         $chval = ord($code{$i});
  893.                         if (($k = array_search($chval, $this->textsubmodes[$submode])) !== false) {
  894.                             // we are on the same sub-mode
  895.                             $txtarr[] = $k;
  896.                         } else {
  897.                             // the sub-mode is changed
  898.                             for ($s = 0; $s < 4; ++$s) {
  899.                                 // search new sub-mode
  900.                                 if (($s != $submode) AND (($k = array_search($chval, $this->textsubmodes[$s])) !== false)) {
  901.                                     // $s is the new submode
  902.                                     if (((($i + 1) == $codelen) OR ((($i + 1) < $codelen) AND (array_search(ord($code{($i + 1)}), $this->textsubmodes[$submode]) !== false))) AND (($s == 3) OR (($s == 0) AND ($submode == 1)))) {
  903.                                         // shift (temporary change only for this char)
  904.                                         if ($s == 3) {
  905.                                             // shift to puntuaction
  906.                                             $txtarr[] = 29;
  907.                                         } else {
